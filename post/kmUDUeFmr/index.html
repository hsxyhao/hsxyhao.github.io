  private installAttach () {
    try {
      this.browser.webContents.debugger.attach('1.1')
    } catch (err) {
      console.log('Debugger attach failed : ', err)
    }
    this.browser.webContents.debugger.on('detach', (event, reason) => {
      console.log('Debugger detached due to : ', reason)
    })
    this.browser.webContents.debugger.on('message', async (event, method, params) => {
      if (method === 'Network.responseReceived') {
        if (params instanceof Object) {
          const res: any = { ...params }
          if (res.response) {
            const url = res.response.url
            if (!url) {
              return
            }
            if (this.siteInterceptor.loginUrl !== '' && url.indexOf(this.siteInterceptor.loginUrl) > 0) {
              const result = await this.browser.webContents.debugger.sendCommand('Network.getResponseBody', { requestId: res.requestId })
              this.loginSuccess(result)
            } else if (this.siteInterceptor.logoutUrl !== '' && url.indexOf(this.siteInterceptor.logoutUrl) > 0) {
              const result = await this.browser.webContents.debugger.sendCommand('Network.getResponseBody', { requestId: res.requestId })
              this.logoutSuccess(result)
            } else if (this.siteInterceptor.balanceUrl !== '' && url.indexOf(this.siteInterceptor.balanceUrl) > 0) {
              const result = await this.browser.webContents.debugger.sendCommand('Network.getResponseBody', { requestId: res.requestId })
              this.balanceSuccess(result)
            } else if (this.siteInterceptor.betUrl !== '' && url.indexOf(this.siteInterceptor.betUrl) > 0) {
              const result = await this.browser.webContents.debugger.sendCommand('Network.getResponseBody', { requestId: res.requestId })
              this.betSuccess(result)
            } else if (this.siteInterceptor.beforeBetUrl !== '' && url.indexOf(this.siteInterceptor.beforeBetUrl) > 0) {
              const result = await this.browser.webContents.debugger.sendCommand('Network.getResponseBody', { requestId: res.requestId })
              this.beforeBetSuccess(result)
            }
          }
        }
      }
    })
    this.browser.webContents.debugger.sendCommand('Network.enable')
  }
